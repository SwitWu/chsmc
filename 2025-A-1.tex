\section{填空题：本大题共 8 小题，每小题 8 分，满分 64 分.}

\begin{enumerate}
  \item 若 $\log_3 (9x)$, $\log_9 (27x)$, $\log_{27} (3x)$
    成等差数列，则正数 $x$ 的值为 \blankline
  \item 设集合 $A = \{1,2,3,\ldots,100\}$, $B = \{a^2+2^a \mid a\in A\}$,
    则 $A\cup B$ 的元素个数为 \blankline
  \item 设点 $P$ 在椭圆 $\Gamma_1: \frac{x^2}{20} + \frac{y^2}{25} = 1$
    上，$F_1,F_2$ 为 $\Gamma_1$ 的两焦点，线段 $F_1P$ 交椭圆
    $\Gamma_2: \frac{x^2}{9} + \frac{y^2}{14} = 1$ 于点 $Q$.
    若 $\triangle F_2PQ$ 的周长为 8，则线段 $F_1Q$ 的长度为 \blankline
  \item 设函数 $f(x)$ 的定义域为 $\mathbb{R}$，$g(x) = (x-1)f(x)$,
    $h(x) = f(x)+x$. 若 $g(x)$ 为奇函数，$h(x)$ 为偶函数，则 $f(x)$ 的最大值为 \blankline
  \item 若正整数 $k$ 满足 $\displaystyle \biggl(
    \frac{\sin 20^\circ}{\cos 25^\circ} 
    + \frac{\sin 25^\circ}{\cos 20^\circ}\cdot\mathrm{i}
    \biggr)^k \in \mathbb{R}$ ($\mathrm{i}$ 为虚数单位)，则 $k$ 的最小值为 \blankline
  \item 设 $\Gamma$ 为任意四棱柱，在 $\Gamma$ 的 12 条棱中随机选取两条不同的棱
    $l_1,l_2$, 将时间“$l_1$ 所在直线与 $l_2$ 所在直线平行”发生的概率记为 $P(\Gamma)$,
    则 $P(\Gamma)$ 的所有可能值为 \blankline
  \item 平面中的 3 个单位向量 $\vec{a}$, $\vec{b}$, $\vec{c}$
    满足 $\vec{a}\cdot\vec{b} = [\vec{a}\cdot\vec{c}] + [\vec{b}\cdot\vec{c}]$
    (其中 $[x]$ 表示不超过实数 $x$ 的最大整数)，则 $|\vec{a}+\vec{b}+\vec{c}|$
    的取值范围是 \blankline
  \item 将 $1,2,3,\ldots,9$ 排列为 $a,b,c,d,e,f,g,h,i$，使得
    3 个三位数 $\overline{abc}$, $\overline{def}$, $\overline{ghi}$
    之和等于 $2025$，则不同的排列方法数为 \blankline
\end{enumerate}

\section{解答题：本大题共 3 个小题，满分 56 分. 解答应写出文字说明、证明过程或演算步骤.}

\begin{enumerate}[resume]
  \item (本题满分 16 分) 在平面直角坐标系 $xOy$ 中，点集
    $\Gamma = \{(x,y) \mid y = 2x+2|x|\}.$
    若 $\Gamma$ 中的 3 个不同的点 $M$, $P$, $Q$ 满足：
    $M$ 为 $PQ$ 的中点，且 $\overrightarrow{OP}\cdot\overrightarrow{OQ} = -2$，
    求点 $M$ 的坐标.
  \item (本题满分 20 分) 设正四面体 $\mathit{ABCD}$ 各棱长均为 $2$，$P$, $Q$
    分别是棱 $AB$, $AC$ 上的动点（允许位于棱的端点），$AP + AQ = 2$，$M$
    为棱 $AD$ 的中点. 在 $\triangle MPQ$ 中， $MH$ 为 $PQ$ 边上的高，
    求 $MH$ 长度的最小值.
  \item (本题满分 20 分) 设 $\alpha$ 为实数，$m$, $n$ 为正整数，
    且 $\sin m\alpha\cdot\sin n\alpha\neq 0$.
    证明：$\displaystyle\frac{1}{|\sin m\alpha|} + \frac{1}{|\sin n\alpha|}
    > \frac{1}{m\cdot |\sin m\alpha\cdot \sin n\alpha| + |\sin mn\alpha|}$.
\end{enumerate}
