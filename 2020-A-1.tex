\section{填空题：本大题共 8 小题，每小题 8 分，共 64 分.}

\begin{enumerate}
  \item 在等比数列 $\{a_n\}$ 中，$a_9 = 13$，$a_{13} = 1$，
    则 $\log_{a_1} 13$ 的值为 \blankline
  \item 在椭圆 $\Gamma$ 中，$A$ 为长轴的一个端点，$B$ 为短轴的一个端点，
    $F_1$, $F_2$ 为两个焦点，若
    $\overrightarrow{AF_1}\cdot\overrightarrow{AF_2} + \overrightarrow{BF_1}\cdot\overrightarrow{BF_2} = 0$，
    则 $\dfrac{|AB|}{|F_1F_2|}$ 的值为 \blankline
  \item 设 $a > 0$，函数 $f(x) = x + \dfrac{100}{x}$ 在区间 $(0,a]$ 上的最小值为 $m_1$，
    在区间 $[a,+\infty)$ 上的最小值为 $m_2$. 若 $m_1 m_2 = 2020$，
    则 $a$ 的值为 \blankline
  \item 设 $z$ 为复数，若 $\dfrac{z-2}{z-\ii}$ 为实数（$\ii$ 为虚数单位），
    则 $|z+3|$ 的最小值为 \blankline
  \item 在 $\triangle ABC$ 中，$AB = 6$，$BC = 4$，边 $AC$ 上的中线长为 $\sqrt{10}$，
    则 $\sin^6\dfrac{A}{2} + \cos^6\dfrac{A}{2}$ 的值为 \blankline
  \item 正三棱锥 $P$-$ABC$ 的所有棱长均为 $1$，$L$, $M$, $N$ 分别为棱
    $PA$, $PB$, $PC$ 的中点，则该正三棱锥的外接球被平面 $LMN$ 所截的截面面积为 \blankline
  \item 设 $a, b > 0$，满足：关于 $x$ 的方程
    $\sqrt{|x|} + \sqrt{|x+a|} = b$ 恰有三个不同的实数解 $x_1$, $x_2$, $x_3$，
    且 $x_1 < x_2 < x_3 = b$，则 $a + b$ 的值为 \blankline
  \item 现有 $10$ 张卡片，每张卡片上写有 $1, 2, 3, 4, 5$ 中两个不同的数，
    且任意两张卡片上的数不完全相同. 将这 $10$ 张卡片放入标号为 $1, 2, 3, 4, 5$
    的五个盒子中，规定写有 $i$, $j$ 的卡片只能放在 $i$ 号或 $j$ 号盒子中.
    一种放法称为“好的”，如果 $1$ 号盒子中的卡片数多于其他每个盒子中的卡片数，
    则“好的”放法共有 \blankline 种.
\end{enumerate}

\section{解答题：本大题共 3 个小题，满分 56 分. 解答应写出文字说明、证明过程或演算步骤.}

\begin{enumerate}[resume]
  \item (本题满分 16 分) 在 $\triangle ABC$ 中，$\sin A = \dfrac{\sqrt{2}}{2}$，
    求 $\cos B + \sqrt{2}\cos C$ 的取值范围.
  \item (本题满分 20 分) 对正整数 $n$ 及实数 $x$ $(0 \leqslant x < n)$，
    定义 $f(n,x) = (1 - \{x\})\cdot\mathrm{C}_n^{[x]} + \{x\}\cdot\mathrm{C}_n^{[x]+1}$，
    其中 $[x]$ 表示不超过实数 $x$ 的最大整数，$\{x\} = x - [x]$.
    若整数 $m, n \geqslant 2$ 满足
    $f\Bigl(m, \dfrac{1}{n}\Bigr) + f\Bigl(m, \dfrac{2}{n}\Bigr)
    + \cdots + f\Bigl(m, \dfrac{mn-1}{n}\Bigr) = 123$，
    求 $f\Bigl(n, \dfrac{1}{m}\Bigr) + f\Bigl(n, \dfrac{2}{m}\Bigr)
    + \cdots + f\Bigl(n, \dfrac{mn-1}{m}\Bigr)$ 的值.
  \item (本题满分 20 分) 在平面直角坐标系中，点 $A$, $B$, $C$ 在双曲线 $xy = 1$ 上，
    满足 $\triangle ABC$ 为等腰直角三角形. 求 $\triangle ABC$ 的面积的最小值.
\end{enumerate}
