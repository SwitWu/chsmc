\section{填空题：本大题共 8 小题，每小题 8 分，共 64 分.}

\begin{enumerate}
  \item 设 $f(x)$ 是定义在 $\mathbf{R}$ 上的函数，对任意实数 $x$ 有
    $f(x+3)\cdot f(x-4) = -1$. 又当 $0 \leqslant x < 7$ 时，
    $f(x) = \log_2(9-x)$，则 $f(-100)$ 的值为 \blankline
  \item 若实数 $x$, $y$ 满足 $x^2 + 2\cos y = 1$，
    则 $x - \cos y$ 的取值范围是 \blankline
  \item 在平面直角坐标系 $xOy$ 中，椭圆 $C$ 的方程为
    $\dfrac{x^2}{9} + \dfrac{y^2}{10} = 1$，$F$ 为 $C$ 的上焦点，
    $A$ 为 $C$ 的右顶点，$P$ 是 $C$ 上位于第一象限内的动点，
    则四边形 $OAPF$ 的面积的最大值为 \blankline
  \item 若一个三位数中任意两个相邻数码的差均不超过 $1$，
    则称其为“平稳数”. 平稳数的个数是 \blankline
  \item 正三棱锥 $P$-$ABC$ 中，$AB = 1$，$AP = 2$，过 $AB$ 的平面 $\alpha$
    将其体积平分，则棱 $PC$ 与平面 $\alpha$ 所成角的余弦值为 \blankline
  \item 在平面直角坐标系 $xOy$ 中，点集
    $K = \{(x,y) \mid x, y = -1, 0, 1\}$. 在 $K$ 中随机取出三个点，
    则这三点中存在两点之间距离为 $\sqrt{5}$ 的概率为 \blankline
  \item 在 $\triangle ABC$ 中，$M$ 是 $BC$ 边的中点，$N$ 是线段 $BM$ 的中点.
    若 $\angle A = \dfrac{\pi}{3}$，$\triangle ABC$ 的面积为 $\sqrt{3}$，
    则 $\overrightarrow{AM}\cdot\overrightarrow{AN}$ 的最小值为 \blankline
  \item 设两个严格递增的正整数数列 $\{a_n\}$, $\{b_n\}$ 满足：
    $a_{10} = b_{10} < 2017$，对任意正整数 $n$，有
    $a_{n+2} = a_{n+1} + a_n$，$b_{n+1} = 2b_n$，
    则 $a_1 + b_1$ 的所有可能值为 \blankline
\end{enumerate}

\section{解答题：本大题共 3 个小题，满分 56 分. 解答应写出文字说明、证明过程或演算步骤.}

\begin{enumerate}[resume]
  \item (本题满分 16 分) 设 $k$, $m$ 为实数，不等式 $|x^2 - kx - m| \leqslant 1$
    对所有 $x \in [a,b]$ 成立. 证明：$b - a \leqslant 2\sqrt{2}$.
  \item (本题满分 20 分) 设 $x_1$, $x_2$, $x_3$ 是非负实数，满足
    $x_1 + x_2 + x_3 = 1$，求
    $(x_1 + 3x_2 + 5x_3)\Bigl(x_1 + \dfrac{x_2}{3} + \dfrac{x_3}{5}\Bigr)$
    的最小值和最大值.
  \item (本题满分 20 分) 设复数 $z_1$, $z_2$ 满足 $\Re(z_1) > 0$，
    $\Re(z_2) > 0$，且 $\Re(z_1^2) = \Re(z_2^2) = 2$
    （其中 $\Re(z)$ 表示复数 $z$ 的实部）.
    \begin{enumerate}[(1)]
      \item 求 $\Re(z_1 z_2)$ 的最小值；
      \item 求 $|z_1 + 2| + |\bar{z}_2 + 2| - |\bar{z}_1 - z_2|$ 的最小值.
    \end{enumerate}
\end{enumerate}
